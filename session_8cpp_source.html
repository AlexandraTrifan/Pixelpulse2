124 int Session::update_available_devices() {
125  m_lock_devlist.lock();
126  m_available_devices.clear();
127  m_lock_devlist.unlock();
128  libusb_device** list;
129  int num = libusb_get_device_list(m_usb_cx, &list);
130  if (num < 0) return num;
131 
132  for (int i=0; i<num; i++) {
133  shared_ptr<Device> dev = probe_device(list[i]);
134  if (dev) {
135  m_lock_devlist.lock();
136  m_available_devices.push_back(dev);
137  m_lock_devlist.unlock();
138  }
139  }
140 
141  libusb_free_device_list(list, true);
142  return 0;
143 }

146 shared_ptr<Device> Session::probe_device(libusb_device* device) {
147  shared_ptr<Device> dev = find_existing_device(device);
148 
149  libusb_device_descriptor desc;
150  int r = libusb_get_device_descriptor(device, &desc);
151  if (r != 0) {
152  cerr << "Error " << r << "in get_device_descriptor" << endl;
153  return NULL;
154  }
155 
156  if (desc.idVendor == 0x59e3 && desc.idProduct == 0xCEE1) {
157  dev = shared_ptr<Device>(new CEE_Device(this, device));
158  } else if (desc.idVendor == 0x0456 && desc.idProduct == 0xCEE2) {
159  dev = shared_ptr<Device>(new M1000_Device(this, device));
160  } else if (desc.idVendor == 0x064B && desc.idProduct == 0x784C) {
161  dev = shared_ptr<Device>(new M1000_Device(this, device));
162  }
163 
164  if (dev) {
165  if (dev->init() == 0) {
166  libusb_get_string_descriptor_ascii(dev->m_usb, desc.iSerialNumber, (unsigned char*)&dev->serial_num, 32);
167  libusb_control_transfer(dev->m_usb, 0xC0, 0x00, 0, 0, (unsigned char*)&dev->m_hw_version, 64, 100);
168  libusb_control_transfer(dev->m_usb, 0xC0, 0x00, 0, 1, (unsigned char*)&dev->m_fw_version, 64, 100);
169 
170  return dev;
171  } else {
172  perror("Error initializing device");
173  }
174  }
175  return NULL;
176 }

238  auto res = m_completion.wait_until(lk, now + std::chrono::milliseconds(1000), [&]{ return m_active_devices == 0; });
