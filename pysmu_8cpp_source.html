53  static PyObject* getDevInfo(PyObject* self, PyObject* args){
54  PyObject* data = PyList_New(0);
55  for (auto i: session->m_available_devices){
56  session->add_device(&*i);
57  }
58  for (auto dev: session->m_devices) {
59  auto dev_info = dev->info();
60  PyObject* dev_data = PyDict_New();
61  for (unsigned chan=0; chan < dev_info->channel_count; chan++) {
62  auto chan_info = dev->channel_info(chan);
63  dev->set_mode(chan, 1);
64  PyObject* sigs = PyList_New(0);
65 
66  for (unsigned sig=0; sig < chan_info->signal_count; sig++) {
67  auto info = dev->signal(chan, sig)->info()->label;
68  PyObject* info_str = PyString_FromString(string(info).c_str());
69  PyList_Append(sigs, info_str);
70  Py_DECREF(info_str);
71  }
72  PyObject* chan_label = PyString_FromString(string(chan_info->label).c_str());
73  PyDict_SetItem(dev_data, chan_label, sigs);
74  Py_DECREF(chan_label);
75  Py_DECREF(sigs);
76  }
77  PyObject* serial_str = PyString_FromString(dev->serial());
78  PyObject* ret = PyTuple_Pack(2, serial_str, dev_data);
79  Py_DECREF(serial_str);
80  Py_DECREF(dev_data);
81  PyList_Append(data, ret);
82  Py_DECREF(ret);
83  }
84  return data;
85  }

143  static PyObject* getAllInputs(PyObject* self, PyObject* args) {
144  const char *dev_serial;
145  int nsamples; /* number of samples to acquire */
146  unsigned num_channels; /* number of channels passed */
147  vector< vector<float> > buf_v, buf_i; /* 2d vector of voltage/current samples per channel */
148 
149  if (!PyArg_ParseTuple(args, "si", &dev_serial, &nsamples))
150  return NULL;
151 
152  auto dev = get_device(dev_serial);
153  if (dev == NULL)
154  return NULL;
155 
156  num_channels = dev->info()->channel_count;
157  buf_v.resize(num_channels);
158  buf_i.resize(num_channels);
159  for (unsigned i = 0; i < num_channels; i++) {
160  buf_v[i].resize(nsamples);
161  buf_i[i].resize(nsamples);
162  dev->signal(i, 0)->measure_buffer(buf_v[i].data(), nsamples);
163  dev->signal(i, 1)->measure_buffer(buf_i[i].data(), nsamples);
164  }
165 
166  session->configure(SAMPLE_RATE);
167  session->run(nsamples);
168 
169  PyObject* all_samples = PyList_New(0);
170  for (unsigned i = 0; i < num_channels; i++) {
171  PyObject* samples = PyList_New(0);
172  for (int j = 0; j < nsamples; j++) {
173  PyObject* sample_tuple = Py_BuildValue("(f,f)", buf_v[i][j], buf_i[i][j]);
174  PyList_Append(samples, sample_tuple);
175  Py_DECREF(sample_tuple);
176  }
177  PyList_Append(all_samples, samples);
178  Py_DECREF(samples);

300  static PyObject* inputs_iternext(inputs *self) {
301  // don't read/pop values from the queues while adding to them
302  std::unique_lock<mutex> lock(signal_mtx);
303 
304  // wait for complete samples to exist in the queue
305  while (signal0_0.empty() || signal0_1.empty() || signal1_0.empty() || signal1_1.empty()) {
306  samples_available.wait(lock);
307  }
308 
309  PyObject* samples_tuple = Py_BuildValue("((f,f),(f,f))",
310  signal0_0.front(), signal0_1.front(), signal1_0.front(), signal1_1.front());
311 
312  signal0_0.pop();
313  signal0_1.pop();
314  signal1_0.pop();
315  signal1_1.pop();
316  lock.unlock();
317 
318  return samples_tuple;
319  }

365  static PyObject * inputs_iter(PyObject *self, PyObject *args) {
366  const char *dev_serial;
367  inputs *p = NULL;
368 
369  if (!PyArg_ParseTuple(args, "s", &dev_serial))
370  return NULL;
371 
372  auto dev = get_device(dev_serial);
373  if (dev == NULL)
374  return NULL;
375 
376  p = PyObject_New(inputs, &inputs_type);
377  if (!p)
378  return NULL;
379 
380  auto signal_callback = [](queue <float> *signal_q, float sample) {
381  std::unique_lock<mutex> lock(signal_mtx);
382  if ((*signal_q).size() < 1024)
383  (*signal_q).push(sample);
384  lock.unlock();
385  if ((signal0_0.size() == 1) && (signal0_1.size() == 1) &&
386  (signal1_0.size() == 1) && (signal1_1.size() == 1))
387  samples_available.notify_one();
388  };
389 
390  // setup signal callbacks
391  dev->signal(0, 0)->measure_callback(std::bind(signal_callback, &signal0_0, _1));
392  dev->signal(0, 1)->measure_callback(std::bind(signal_callback, &signal0_1, _1));
393  dev->signal(1, 0)->measure_callback(std::bind(signal_callback, &signal1_0, _1));
394  dev->signal(1, 1)->measure_callback(std::bind(signal_callback, &signal1_1, _1));
395 
396  session->configure(SAMPLE_RATE);
397  // run in continuous mode
398  session->start(0);
399 
400  return (PyObject *)p;
401  }
