304  inline void put_sample(value_t val) {
305  m_latest_measurement = val;

317  inline value_t get_sample() {
318  switch (m_src) {
319  case SRC_CONSTANT:
320  return m_src_v1;
321 
322  case SRC_BUFFER:
323  if (m_src_i >= m_src_buf_len) {
324  if (m_src_buf_repeat) {
325  m_src_i = 0;
326  } else {
327  return m_src_buf[m_src_buf_len-1];
328  }
329  }
330  return m_src_buf[m_src_i++];
331 
332 
333  case SRC_CALLBACK:
334  return m_src_callback(m_src_i++);
335 
336  case SRC_SQUARE:
337  case SRC_SAWTOOTH:
338  case SRC_SINE:
339  case SRC_STAIRSTEP:
340  case SRC_TRIANGLE:
341 
342  auto pkpk = m_src_v2 - m_src_v1;
343  auto phase = m_src_phase;
344  auto norm_phase = phase / m_src_period;
345  m_src_phase = fmod(m_src_phase + 1, m_src_period);
346 
347  switch (m_src) {
348  case SRC_SQUARE:
349  return (norm_phase < m_src_duty) ? m_src_v1 : m_src_v2;
350 
351  case SRC_SAWTOOTH:
352  return m_src_v1 + norm_phase * pkpk;
353 
354  case SRC_STAIRSTEP:
355  return m_src_v2 - floorf(norm_phase*10)/10 * pkpk;
356 
357  case SRC_SINE:
358  return m_src_v1 + (1 + cos(norm_phase * 2 * M_PI)) * pkpk/2;
359 
360  case SRC_TRIANGLE:
361  return m_src_v1 + fabs(1 - norm_phase*2) * pkpk;
362  default:
363  return 0;
364  }
365  }
366  return 0;
367  }
