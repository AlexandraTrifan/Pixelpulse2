106 void M1000_Device::in_completion(libusb_transfer *t) {
107 
108  std::lock_guard<std::mutex> lock(m_state);
109  m_in_transfers.num_active--;
110 
111  if (t->status == LIBUSB_TRANSFER_COMPLETED) {
112  handle_in_transfer(t);
113  // m_cancellation == 0, everything OK
114  if (m_session->m_cancellation == 0) {
115  submit_in_transfer(t);
116  }
117  } else if (t->status != LIBUSB_TRANSFER_CANCELLED) {
118  m_session->handle_error(t->status, "M1000_Device::in_completion");
119  }
120  if (m_out_transfers.num_active == 0 && m_in_transfers.num_active == 0) {
121  m_session->completion();
122  }

153 void M1000_Device::configure(uint64_t rate) {
154  double sample_time = 1.0/rate;
155  double M1K_timer_clock;
156  // if FW version is 023314a - initial production, use 3e6 for timer clock
157  if (strcmp(this->m_fw_version, "023314a*") == 0) {
158  M1K_timer_clock = 3e6;
159  }
160  // otherwise, assume a more recent firmware, and use the audacious clock
161  else {
162  M1K_timer_clock = 48e6;
163  }
164  m_sam_per = round(sample_time * M1K_timer_clock) / 2;
165  if (m_sam_per < m_min_per) m_sam_per = m_min_per;
166  sample_time = m_sam_per / M1K_timer_clock; // convert back to get the actual sample time;
167 
168  unsigned transfers = 8;
169  m_packets_per_transfer = ceil(BUFFER_TIME / (sample_time * chunk_size) / transfers);
170 
171  m_in_transfers.alloc( transfers, m_usb, EP_IN, LIBUSB_TRANSFER_TYPE_BULK,
172  m_packets_per_transfer*in_packet_size, 10000, m1000_in_completion, this);
173  m_out_transfers.alloc(transfers, m_usb, EP_OUT, LIBUSB_TRANSFER_TYPE_BULK,
174  m_packets_per_transfer*out_packet_size, 10000, m1000_out_completion, this);
175  m_in_transfers.num_active = m_out_transfers.num_active = 0;
176  //std::cerr << "M1000 rate: " << sample_time << " " << m_sam_per << std::endl;
177 }

180 inline uint16_t M1000_Device::encode_out(unsigned chan) {
181  int v = 0;
182  if (m_mode[chan] == SVMI) {
183  float val = m_signals[chan][0].get_sample();
184  val = constrain(val, 0, 5.0);
185  v = 65535*val/5.0;
186  } else if (m_mode[chan] == SIMV) {
187  float val = m_signals[chan][1].get_sample();
188  val = constrain(val, -current_limit, current_limit);
189  v = 65536*(2./5. + 0.8*0.2*20.*0.5*val);
190  } else if (m_mode[chan] == DISABLED) {
191  v = 32768*4/5;
192  }
193  if (v > 65535) v = 65535;
194  if (v < 0) v = 0;
195  return v;
196 }

199 bool M1000_Device::submit_out_transfer(libusb_transfer* t) {
200  if (m_sample_count == 0 || m_out_sampleno < m_sample_count) {
201  for (unsigned p=0; p<m_packets_per_transfer; p++) {
202  uint8_t* buf = (uint8_t*) (t->buffer + p*out_packet_size);
203  for (unsigned i=0; i < chunk_size; i++) {
204  uint16_t a = encode_out(0);
205  buf[(i+chunk_size*0)*2 ] = a >> 8;
206  buf[(i+chunk_size*0)*2+1] = a & 0xff;
207  uint16_t b = encode_out(1);
208  buf[(i+chunk_size*1)*2 ] = b >> 8;
209  buf[(i+chunk_size*1)*2+1] = b & 0xff;
210  m_out_sampleno++;
211  }
212  }
213  int r = libusb_submit_transfer(t);
214  if (r != 0) {
215  m_out_transfers.failed(t);
216  // writes to t->status is illegal
217  // t->status = (libusb_transfer_status) r;
218  m_session->handle_error(r, "M1000_Device::submit_out_transfer");
219  return false;
220  }
221  m_out_transfers.num_active++;
222  return true;
223  }
224  return false;
225 }

230  if (m_sample_count == 0 || m_requested_sampleno < m_sample_count) {
231  int r = libusb_submit_transfer(t);
232  if (r != 0) {

246 void M1000_Device::handle_in_transfer(libusb_transfer* t) {
247  for (unsigned p=0; p<m_packets_per_transfer; p++) {
248  uint8_t* buf = (uint8_t*) (t->buffer + p*in_packet_size);
249 
250  for (unsigned i=(m_in_sampleno==0)?2:0; i<chunk_size; i++) {
251  m_signals[0][0].put_sample( (buf[(i+chunk_size*0)*2] << 8 | buf[(i+chunk_size*0)*2+1]) / 65535.0 * 5.0);
252  m_signals[0][1].put_sample((((buf[(i+chunk_size*1)*2] << 8 | buf[(i+chunk_size*1)*2+1]) / 65535.0 * 0.4 ) - 0.195)*1.25);
253  m_signals[1][0].put_sample( (buf[(i+chunk_size*2)*2] << 8 | buf[(i+chunk_size*2)*2+1]) / 65535.0 * 5.0);
254  m_signals[1][1].put_sample((((buf[(i+chunk_size*3)*2] << 8 | buf[(i+chunk_size*3)*2+1]) / 65535.0 * 0.4) - 0.195)*1.25);
255  m_in_sampleno++;
256  }
257  }
258 
259  m_session->progress();
260 }

284 void M1000_Device::set_mode(unsigned chan, unsigned mode) {
285  if (chan < 2) {
286  m_mode[chan] = mode;
287  }
288  // set feedback potentiometers with mode heuristics
289  unsigned pset;
290  switch (mode ) {

293  case DISABLED:

297  // set mode
