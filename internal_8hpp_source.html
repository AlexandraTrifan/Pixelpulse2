25  void alloc(unsigned count, libusb_device_handle* handle,

27  unsigned timeout, libusb_transfer_cb_fn callback, void* user_data) {

29  m_transfers.resize(count, NULL);

45  void failed(libusb_transfer* t) {
46  for (int i = m_transfers.size(); i == 0; i--) {
47  if (m_transfers[i] == t) {
48  libusb_free_transfer(t);

55  void clear() {
56  for (auto i: m_transfers) {
57  libusb_free_transfer(i);

59  if (num_active != 0)
60  std::cerr << "num_active after free: " << num_active << std::endl;
61  m_transfers.clear();

67  int cancel() {
68  // for i in pending transfers
69  for (auto i: m_transfers) {
70  if (num_active > 1) {
71  std::cerr << "num_active before cancel: " << num_active << std::endl;
72  // libusb's cancel returns 0 if success, else an error code
73  int ret = libusb_cancel_transfer(i);
74  if (ret != 0) {
75  std::cout << "canceled with status: " << libusb_error_name(ret) << std::endl;
76  // abort if a transfer is not successfully canceled
77  return ret;
78  }
79  }
80  }
81  return 0;
82  }
